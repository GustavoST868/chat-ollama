        function addMessageToCurrentChat(role, content) {
            if(!currentChatId) currentChatId = createNewChat();
            const chat = chats[currentChatId];
            chat.messages.push({ role, content, timestamp: new Date().toISOString() });
            chat.updatedAt = new Date().toISOString();
            if(chat.messages.length === 1 && role === 'user') {
                chat.title = content.slice(0, 30) + (content.length > 30 ? 'âŠ' : '');
            }
            saveChatsToStorage();
            renderChatMessages(currentChatId);
            renderChatsList();
        }

        function updateLastAssistantMessage(content) {
            if(!currentChatId) return;
            const chat = chats[currentChatId];
            const last = chat.messages[chat.messages.length-1];
            if(last && last.role === 'assistant') last.content = content;
            else chat.messages.push({ role: 'assistant', content, timestamp: new Date().toISOString() });
            saveChatsToStorage();
            renderChatMessages(currentChatId);
        }

        function deleteChat(chatId) {
            if(chatId === currentChatId) {
                const newId = createNewChat();
                loadChat(newId);
            }
            delete chats[chatId];
            saveChatsToStorage();
            renderChatsList();
            if(Object.keys(chats).length === 0) {
                const newId = createNewChat();
                loadChat(newId);
            } else if(currentChatId === chatId) {
                const firstId = Object.keys(chats)[0];
                loadChat(firstId);
            }
        }

        // ---------- Streaming request ----------
        async function sendStreamingRequest(model, payload) {
            currentAbortController = new AbortController();
            const response = await fetch(GENERATE_URL, {
                method: "POST",
                headers: { "Content-Type": "application/json" },
                body: JSON.stringify({ 
                    model, 
                    system: payload.system,
                    prompt: payload.prompt, 
                    stream: true, 
                    options: { temperature: 0, seed: 42 } 
                }),
                signal: currentAbortController.signal
            });
            if(!response.ok) throw new Error(`HTTP ${response.status}`);
            const reader = response.body.getReader();
            const decoder = new TextDecoder("utf-8");
            let buffer = "", accumulated = "";
            while(true) {
                const { done, value } = await reader.read();
                if(done) break;
                buffer += decoder.decode(value, { stream: true });
                const lines = buffer.split("\n");
                buffer = lines.pop();
                for(const line of lines) {
                    if(!line.trim()) continue;
                    try {
                        const chunk = JSON.parse(line);
                        if(chunk.response) {
                            accumulated += chunk.response;
                            if(currentStreamBubble) {
                                currentStreamBubble.innerHTML = accumulated + '<span class="streaming-cursor"></span>';
                                scrollToBottom();
                            }
                        }
                    } catch(e) {}
                }
            }
            return accumulated;
        }

        // ---------- Voice Call Logic ----------
        function initSpeechRecognition() {
            const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
            if(!SpeechRecognition) {
                alert("Seu navegador nĂŁo suporta reconhecimento de voz.");
                return null;
            }
            const rec = new SpeechRecognition();
            rec.lang = callLanguage;
            rec.continuous = false;
            rec.interimResults = true;

            rec.onstart = () => {
                callStatus.innerText = "Ouvindo...";
                callVisualizer.classList.add('listening');
                callVisualizer.classList.remove('speaking');
            };

            rec.onresult = (event) => {
                let interimTranscript = '';
                let finalTranscript = '';
                for (let i = event.resultIndex; i < event.results.length; ++i) {
                    if (event.results[i].isFinal) finalTranscript += event.results[i][0].transcript;
                    else interimTranscript += event.results[i][0].transcript;
                }
                callTranscription.innerText = finalTranscript || interimTranscript;
            };

            rec.onend = async () => {
                if(!isCallActive || isMuted) return;
                const text = callTranscription.innerText.trim();
                if(text) {
                    await handleVoiceInput(text);
                } else {
                    // Se nĂŁo ouviu nada, volta a ouvir apĂłs um pequeno delay
                    setTimeout(() => { if(isCallActive && !isMuted && !synthesis.speaking) try { rec.start(); } catch(e){} }, 400);
                }
            };

            rec.onerror = (event) => {
                console.error("Speech error", event.error);
                if(isCallActive) setTimeout(() => { try { rec.start(); } catch(e){} }, 1000);
            };

            return rec;
        }

        async function handleVoiceInput(text) {
            callStatus.innerText = "Processando...";
            callVisualizer.classList.remove('listening');
            
            // Adiciona ao chat (silenciosamente ou nĂŁo)
            addMessageToCurrentChat('user', text);
            
            const selectedModel = modelSelector.value;
            const fullPrompt = buildFullPromptWithHistory(text);

            try {
                const response = await sendStreamingRequest(selectedModel, fullPrompt);
                updateLastAssistantMessage(response);
                speak(response);
            } catch(err) {
                console.error(err);
                callStatus.innerText = "Erro na resposta.";
                if(isCallActive) setTimeout(() => recognition.start(), 2000);
            }
        }

        function speak(text) {
            if(!isCallActive) return;
            
            // Remove markdown e caracteres especiais para uma leitura melhor
            const cleanText = text.replace(/```[\s\S]*?```/g, 'CĂłdigo.')
                                 .replace(/[*_#]/g, '')
                                 .trim();

            const utterance = new SpeechSynthesisUtterance(cleanText);
            
            if (selectedVoice) {
                utterance.voice = selectedVoice;
            } else {
                utterance.lang = callLanguage;
            }

            utterance.rate = 0.95; // Ligeiramente mais lento soa mais natural
            utterance.pitch = 1.0;

            utterance.onstart = () => {
                callStatus.innerText = "Assistente falando...";
                callVisualizer.classList.add('speaking');
                callVisualizer.classList.remove('listening');
            };

            utterance.onend = () => {
                if(isCallActive && !isMuted) {
                    callTranscription.innerText = "";
                    try { recognition.start(); } catch(e){}
                }
            };

            synthesis.speak(utterance);
        }

        function startCall() {
            isCallActive = true;
            callOverlay.classList.add('active');
            callTranscription.innerText = "";
            recognition = initSpeechRecognition();
            if(recognition) recognition.start();
        }

        function endCall() {
            isCallActive = false;
            if(recognition) {
                try { recognition.stop(); } catch(e){}
                recognition = null;
            }
            synthesis.cancel();
            callOverlay.classList.remove('active');
        }

        // ---------- Voice Management ----------
        function updateVoices() {
            voices = synthesis.getVoices();
            voiceSelector.innerHTML = "";
            
            // Filtra vozes em portuguĂȘs ou do Google (que costumam ser melhores)
            const ptVoices = voices.filter(v => v.lang.includes('pt') || v.lang.includes('PT'));
            
            if (ptVoices.length === 0) {
                const opt = document.createElement('option');
                opt.textContent = "Nenhuma voz PT encontrada";
                voiceSelector.appendChild(opt);
                return;
            }

            ptVoices.forEach(voice => {
                const option = document.createElement('option');
                option.textContent = `${voice.name} (${voice.lang})`;
                option.value = voice.name;
                voiceSelector.appendChild(option);
            });

            // Tenta selecionar uma voz do Google por padrĂŁo (costumam ser as melhores no Chrome)
            const googlePt = ptVoices.find(v => v.name.includes('Google') && v.lang.includes('pt-BR'));
            const defaultVoice = googlePt || ptVoices[0];
            
            if (defaultVoice) {
                voiceSelector.value = defaultVoice.name;
                selectedVoice = defaultVoice;
            }
        }

        // O evento voiceschanged Ă© necessĂĄrio porque as vozes sĂŁo carregadas de forma assĂ­ncrona
        if (speechSynthesis.onvoiceschanged !== undefined) {
            speechSynthesis.onvoiceschanged = updateVoices;
        }

        voiceSelector.addEventListener('change', () => {
            selectedVoice = voices.find(v => v.name === voiceSelector.value);
        });

                callStatus.innerText = "Ouvindo...";
            }
        }

    })();
